def LMSR(q0, q1, b):
    return b * np.log(np.power(np.e, (q0 / b)) + np.power(np.e, (q1 / b)))

def LSLMSR(q0, q1, a):
    b = a * (q0 + q1)
    return LMSR(q0,q1,b)
